THE NEURAL
STABILIZER LOGIC

We implemented a hierarchical reinforcement learning (HRL) architecture. The "Commander" layer sets long-term environmental targets, while the "Worker" layer manages millisecond-level valve adjustments.

class MartianNet(nn.Module):
    def __init__(self, input_dim=512):
        super().__init__()
        self.encoder = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(d_model=input_dim, nhead=8),
            num_layers=6
        )
        self.policy_head = nn.Linear(input_dim, 24)
        
    def forward(self, telemetry):
        features = self.encoder(telemetry)
        return torch.softmax(self.policy_head(features), dim=-1)
